import os
def count_char_occurrences(char, file_path):
    if len(char) != 1:
        return "Ошибка: символ должен быть одиночным символом"

    if not os.path.exists(file_path):
        return "Ошибка: указанный файл не существует"

    count = 0
    with open(file_path, "r") as file:
        text = file.read()
        for c in text:
            if c == char:
                count += 1

    return count


# Пример использования
char = "a"
file_path = "sample.txt"
result = count_char_occurrences(char, file_path)
print(f"Символ '{char}' встречается {result} раз в файле {file_path}")
